    <script>
        /* --------------------------------------------------
           1. PUBLISHED POEMS DATA STRUCTURE
           -------------------------------------------------- */
        const publishedPoems = [
            {
                title: "Paper Constellations",
                journal: "The Literary Review",
                year: "2024",
                symbol: "✦",
                size: "3.5rem",
                top: "22%",
                left: "18%"
            },
            {
                title: "Notes on Static",
                journal: "Void Journal",
                year: "2025",
                symbol: "✧",
                size: "2.8rem",
                top: "32%",
                left: "78%"
            },
            {
                title: "Midnight Syntax",
                journal: "Polyphony Lit",
                year: "2025",
                symbol: "✩",
                size: "3.2rem",
                top: "68%",
                left: "82%"
            },
            {
                title: "Quiet Geography",
                journal: "Adroit Journal",
                year: "2026",
                symbol: "✶",
                size: "3.8rem",
                top: "72%",
                left: "25%"
            },
            {
                title: "Acoustic Horizon",
                journal: "Rust & Moth",
                year: "2025",
                symbol: "✷",
                size: "2.5rem",
                top: "55%",
                left: "52%"
            }
        ];

        /* --------------------------------------------------
           2. RENDER POEM STARS
           -------------------------------------------------- */
        const starsContainer = document.getElementById('poem-stars-container');

        publishedPoems.forEach((poem, index) => {
            const starEl = document.createElement('div');
            starEl.className = 'poem-star';
            starEl.style.top = poem.top;
            starEl.style.left = poem.left;
            starEl.style.setProperty('--star-size', poem.size);
            starEl.setAttribute('data-index', index);
            starEl.innerHTML = `
                ${poem.symbol}
                <div class="star-tooltip">
                    <span class="tooltip-title">${poem.title}</span>
                    <span class="tooltip-journal">Published in ${poem.journal} (${poem.year})</span>
                </div>
            `;
            starsContainer.appendChild(starEl);
        });

        /* --------------------------------------------------
           3. GENERATE AMBIENT ATMOSPHERE (LAYER 1)
           -------------------------------------------------- */
        const sky = document.getElementById('sky');
        const ambientCount = 70;

        for (let i = 0; i < ambientCount; i++) {
            const star = document.createElement('div');
            star.className = 'ambient-star';
            const size = Math.random() * 2 + 1 + 'px';
            star.style.width = size;
            star.style.height = size;
            star.style.top = Math.random() * 100 + '%';
            star.style.left = Math.random() * 100 + '%';
            star.style.setProperty('--duration', (Math.random() * 3 + 3) + 's');
            sky.appendChild(star);
        }

        /* --------------------------------------------------
           4. NEAREST-NEIGHBOR CONSTELLATION LINES
           -------------------------------------------------- */
        const canvas = document.getElementById('constellation-canvas');
        const ctx = canvas.getContext('2d');
        const poemStarElements = document.querySelectorAll('.poem-star');

        function resizeCanvas() {
            canvas.width = sky.clientWidth;
            canvas.height = sky.clientHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        function getStarCenter(el) {
            const rect = el.getBoundingClientRect();
            const skyRect = sky.getBoundingClientRect();
            return {
                x: rect.left - skyRect.left + rect.width / 2,
                y: rect.top - skyRect.top + rect.height / 2
            };
        }

        poemStarElements.forEach(activeStar => {
            activeStar.addEventListener('mouseenter', () => {
                const activePos = getStarCenter(activeStar);
                
                // 1. Calculate distances from active star to all other stars
                const distances = [];
                poemStarElements.forEach(otherStar => {
                    if (otherStar !== activeStar) {
                        const otherPos = getStarCenter(otherStar);
                        const dx = otherPos.x - activePos.x;
                        const dy = otherPos.y - activePos.y;
                        const dist = Math.sqrt(dx * dx + dy * dy);
                        distances.push({ pos: otherPos, dist: dist });
                    }
                });

                // 2. Sort by distance and pick the 2 closest neighbors
                distances.sort((a, b) => a.dist - b.dist);
                const nearestNeighbors = distances.slice(0, 2);

                // 3. Draw subtle connecting constellation lines
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                ctx.strokeStyle = 'rgba(255, 255, 255, 0.25)';
                ctx.lineWidth = 1;
                ctx.setLineDash([4, 4]); // Delicate dashed constellation feel

                nearestNeighbors.forEach(neighbor => {
                    ctx.beginPath();
                    ctx.moveTo(activePos.x, activePos.y);
                    ctx.lineTo(neighbor.pos.x, neighbor.pos.y);
                    ctx.stroke();
                });
            });

            activeStar.addEventListener('mouseleave', () => {
                ctx.clearRect(0, 0, canvas.width, canvas.height);
            });
        });
    </script>
